  <script>
    function track() {
      let code = document.getElementById("track").value;
      let result = document.getElementById("result");

      if (code === "FX123") {
        result.innerHTML = "Status: Pending 🟡";
        result.className = "result pending";
      } 
      else if (code === "FX456") {
        result.innerHTML = "Status: In Transit 🔵";
        result.className = "result transit";
      } 
      else if (code === "FX789") {
        result.innerHTML = "Status: Delivered 🟢";
        result.className = "result delivered";
      } 
      else {
        result.innerHTML = "Tracking Number Not Found ❌";
        result.className = "result notfound";
      }
    }
  </script>
